    public static void insert(int arr[],int len){
        if(len <=1) return;
        for (int i = 1; i < len; i++) {
            int value = arr[i];
            int j = i - 1;
            for (; j >= 0; j--) {
                if(arr[j] > value){
                    arr[j + 1] = arr[j];
                }else {
                    break;
                }
            }
            arr[j + 1] = value;
        }
    }
    public static void choose(int arr[],int len){
        if(len <=1) return;
        for (int i = 0; i < len; i++) {
            int min = i; //
            for (int j = i; j < len; j++) {
                if(arr[j] < arr[min]){
                    min = j;
                }
            }
            int tmp = arr[i];
            arr[i] = arr[min];
            arr[min] = tmp;
        }

    }

    public static void bubble(int arr[],int len){
        for (int i = 0; i < len; i++) {
            boolean flg = false;
            for (int j = 0; j <  len - i - 1; j++) {
                if(arr[j] > arr[j+1]){
                    int tmp = arr[j];
                    arr[j] = arr[j+1];
                    arr[j+1] = tmp;
                    flg  = true;
                }
            }
            if(!flg){
                break;
            }
        }
    }
